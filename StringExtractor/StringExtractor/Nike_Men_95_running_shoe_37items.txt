  /**
   * Gets the value of a cookie by cookie name
   */
  function getCookie(name) {
    var allCookies = document.cookie.split( ';' );
    var tempCookie = '';
    var cookieName = '';
    var cookieValue = '';
    var cookieFound = false; // set boolean t/f default f
    var i;

    for ( i = 0; i < allCookies.length; i++ ){
      tempCookie = allCookies[i].split( '=' );
      cookieName = tempCookie[0].replace(/^\s+|\s+$/g, '');
      if ( cookieName == name ){
        cookieFound = true;
        // we need to handle case where cookie has no value but exists (no = sign, that is):
        if ( tempCookie.length > 1 ){
          cookieValue = unescape( tempCookie[1].replace(/^\s+|\s+$/g, '') );
        }
        return cookieValue;
        break;
      }
      tempCookie = null;
      cookieName = '';
    }

    if ( !cookieFound ){
      return null;
    }
  }

  /**
   * Get the queryString for the current url and wrap in the referrer as the 'ref' param if it hasn't been added already.

  function buildQueryString(){
    var retQS = window.location.search,
        ref;

    //Setup ref param to capture referrer if the query does not already have a ref param
    if(!/(^|&)ref=/.test(window.location.search)){
      ref = encodeURIComponent(document.referrer);
      // removes referrer if root domain is present in domain name
      var rootDomain = window.location.hostname;
      var hostNameParts = rootDomain.split(".");
      ref = ref.indexOf('/', 8) > -1 ? ref.slice(0, ref.indexOf('/', 8)) : ref;
      if(hostNameParts.length > 1) {
        var rootDomainPart = hostNameParts.slice(hostNameParts.length - 2);
        rootDomain = rootDomainPart ? "." + rootDomainPart.join(".") : rootDomain;
      }
      if(ref && ref.indexOf(rootDomain) > -1) {
        ref = "";
      }
    }

    if(ref && ref.length  > 0){
      if(retQS && retQS.length > 0){
        retQS += "&ref=" + ref;
      } else{
        retQS = "?ref=" + ref;
      }
    }
    return retQS;
  }
